    <script>
        // Toggle sidebar collapse
        const toggleButton = document.getElementById('toggleSidebar');
        const sidebar = document.getElementById('sidebar');
        const mainContent = document.getElementById('mainContent');

        toggleButton.addEventListener('click', () => {
            sidebar.classList.toggle('collapsed');
            mainContent.classList.toggle('collapsed');
            toggleButton.innerHTML = sidebar.classList.contains('collapsed') ? '<i class="fas fa-chevron-right"></i>' : '<i class="fas fa-chevron-left"></i>';
        });

        // Ensure the default theme is light
        document.body.classList.remove('dark-theme');

        // New functionality for theme selection
        document.getElementById('lightTheme').addEventListener('click', () => {
            document.body.classList.remove('dark-theme');
            const themeModal = bootstrap.Modal.getInstance(document.getElementById('themeModal'));
            themeModal.hide();
        });

        document.getElementById('darkTheme').addEventListener('click', () => {
            document.body.classList.add('dark-theme');
            const themeModal = bootstrap.Modal.getInstance(document.getElementById('themeModal'));
            themeModal.hide();
        });

        function navigateToDetail(name, image) {
            window.location.href = `personDetail.html?name=${encodeURIComponent(name)}&image=${encodeURIComponent(image)}`;
        }

        document.getElementById('logoutButton').addEventListener('click', () => {
            localStorage.removeItem('loggedIn'); // Clear login state
            window.location.href = 'signin.html'; // Navigate to sign-in page
        });

        // Check if the user is logged in
        window.onload = function() {
            if (!localStorage.getItem('loggedIn')) {
                window.location.href = 'signin.html'; // Redirect to sign-in if not logged in
            }
        };

        // Hide sidebar on scroll
        window.addEventListener('scroll', () => {
            if (window.scrollY + window.innerHeight >= document.body.offsetHeight) {
                sidebar.classList.add('hidden'); // Hide sidebar when scrolling to footer
            } else {
                sidebar.classList.remove('hidden'); // Show sidebar when not at footer
            }
        });
    </script>
